            function validate() {
                var c;
                var id = document.getElementById("id");
                var value = id.value;

                // Validation of ID field
                // ####/yyyy
                if(value.length != 9) {
                    alert("Invalid Serial Number\nThe length is less than 9");
                    id.focus();
                    return false;
                } else {

                }

                for(let i=0; i<=9; i++) {
                    c = value.charCodeAt(i);
                    if(i>=0 && i<=3) {
                        if( !(c>=48 || c<57) ) {
                            alert("Invalid Serial Number\nFirst 4 digits are not correct");
                            id.focus();
                            return false;
                        }
                    } else if(i==4) {
                        if(c!==47) {
                            alert("Invalid Student ID\nThe 5th position is not slash (/) "+c);
                            id.focus();
                            return false;
                        }
                    } else if(i>=5 && i<=8) {
                        if( !(c>=48 && c<57) ) {
                            alert("Invalid Year\nLast 4 digits must be the year");
                            id.focus();
                            return false;
                        }
                    }
                }

                if(isNaN(value.substring(0,4))) {
                    alert("Not a Number!");
                    id.focus();
                    return false;
                }

                if(isNaN(value.substring(5))) {
                    alert("Not a Year!" + value.substr(5));
                    id.focus();
                    return false;
                }

                // Validation of NAME field
                var name = document.getElementById("name");
                value = name.value;
                alert(value);
                for(let i=0; i<value.length; i++) {
                    c = value.charCodeAt(i);
                    if( c!=32 || !(c>=97 && c<=90) || !(c>=97 && c<=122) ) {
                        alert("Special Character in Name!\nNot allowed.");
                        name.focus();
                        return false;
                    } else {
                        document.write(c);
                    }
                }

                return true;
            }
        </script>
